						limit = { owner = { is_machine_empire = yes } }
						add_building = building_machine_capital
					}
					
					create_pop = {
						species = prev.owner_main_species
					}
					
					solar_system = {
						if = {
							limit = { exists = starbase }
							starbase = {
								set_owner = prev.owner
							}
						}
						else = {
							create_starbase = {
								size = starbase_starport
								module = shipyard
								owner = prev.owner
							}
						}
					}
				}
			}
			
			while = {
				count = 10
				give_random_technology_hidden = yes
			}
			
			every_owned_planet = {
				add_deposit = d_sarcophagus
			}
		}
